class packet;
  randc int a[100]; // Fixed-size array of 100 integers
  function void pre_randomize();
    foreach (a[i])
      a[i] = i + 1; // Fill with 1 to 100
  endfunction
  function void post_randomize();
    a.shuffle(); // Shuffle to make it random without repetition
  endfunction
endclass
module test;
  packet p1;
  initial
    begin
    p1 = new();
    if (p1.randomize())
      begin
      $display("100 Unique Random Numbers:%p\n", p1.a);
    end
    else
      begin
      $display("Randomization failed");
    end
  end
endmodule
